# React 프로젝트 생성 도구 설치
- npm i -g create-react-app

# React 프로젝트 생성 및 실행
- create-react-app [projectName]  : project 생성
- cd [projectName]                : project 경로 이동
- npm start                       : local 환경에서 실행

# Redux 설치
- npm i redux
- npm i redux-saga
- npm i redux-actions
- npm i react-redux

# Firebase 설치
- npm i -g firebase-tools
- npm i firebase

# Firebase 연결
- firebase login
- firebase init

# Firebase 호스팅 배포
- npm run build
- firebase deploy

# Firebase 호스팅 테스트
- firebase serve
